
### 1. Fibonacci Series

###### Recursive
```java
f(int n)
{
	// base case
	if (n == 0 or n == 1)
	{
		return n;
	}	
	return f(n-1) + f(n-2);
}
```

> Dimensions of DP array = no. of changing variables


###### Memoization
```java
Integer[] dp;
f(int n)
{
	// base case
	if (n == 0 or n == 1)
	{
		return n;
	}	
	if (dp[n] != null)
	{
		return dp[n];
	}
	return dp[n] = f(n-1) + f(n-2);
}
```


###### Bottom Up
```java
class Solution {
    Integer[] dp;
    public int fib(int n) {
        dp = new Integer[n+1];
        return f(n);
    }

    private int f(int n)
    {
        // base case
        if (n <= 1)
        {
            return n;
        }
        dp[0] = 0;
        dp[1] = 1;
        
        for (int i = 2; i <= n; i++)
        {
            dp[i] = dp[i-1] + dp[i-2];
        }

        return dp[n];
    }
}
```

###### Constant Space
```java
class Solution {
    public int fib(int n) {
        return f(n);
    }

    private int f(int n)
    {
        // base case
        if (n <= 1)
        {
            return n;
        }
        int prev1 = 0;
        int prev2 = 1;
        
        for (int i = 2; i <= n; i++)
        {
            int curr = prev1 + prev2;
            prev1 = prev2;
            prev2 = curr;
        }

        return prev2;
    }
}
```


### 2. [Climbing Stairs](https://leetcode.com/problems/climbing-stairs/)

###### Memoization - Top Down

```java
class Solution {
    Integer[] dp;
    public int climbStairs(int n) {
        dp = new Integer[n+1];
        return solve(n);
    }

    private int solve(int n)
    {
        // base case
        if (n < 0)  return 0;
        if (n == 0) return 1;
        if (dp[n] != null)
        {
            return dp[n];
        }

        int s1 = solve(n-1);
        int s2 = solve(n-2);

        return dp[n] = s1 + s2;
    }
}
```

###### Bottom Up

```java
class Solution {
    Integer[] dp;
    public int climbStairs(int n) {
        dp = new Integer[n+1];
        return solve(n);
    }

    private int solve(int n)
    {
        // base case
        if (n == 0 || n == 1 || n == 2)
            return n;

        dp[0] = 0;
        dp[1] = 1;
        dp[2] = 2;

        for (int i = 3; i <= n; i++)
        {
            dp[i] = dp[i-1] + dp[i-2];
        }
        return dp[n];
    }
}
```

## 3. [House Robber](https://leetcode.com/problems/house-robber/)
###### Recursive
```java
class Solution {
    public int rob(int[] nums) {
        return solve(0, nums);
    }
    private int solve(int i, int[] nums)
    {
        // base case
        if (i >= nums.length)
        {
            return 0;
        }
        int steal = nums[i] + solve(i+2, nums);
        int skip = solve(i+1, nums);

        return Math.max(steal, skip);
    }
}
```
###### Memoization
```java
class Solution {
    Integer[] dp;
    public int rob(int[] nums) {
        dp = new Integer[nums.length+1];
        return solve(0, nums);
    }
    private int solve(int i, int[] nums)
    {
        // base case
        if (i >= nums.length)
        {
            return 0;
        }
        if (dp[i] != null)
        {
            return dp[i];
        }
        int steal = nums[i] + solve(i+2, nums);
        int skip = solve(i+1, nums);

        return dp[i] = Math.max(steal, skip);
    }
}
```
###### Bottom Up
```java
class Solution {
    Integer[] dp;
    public int rob(int[] nums) {
        dp = new Integer[nums.length+1];
        return solve(nums);
    }
    private int solve(int[] nums)
    {
        // base case
        if (nums.length == 1)
        {
            return nums[0];
        }

        dp[0] = 0;
        dp[1] = nums[0];

        for (int i = 2; i<= nums.length; i++)
        {
            int steal = nums[i-1] + dp[i-2];
            int skip = dp[i-1];
            dp[i] = Math.max(steal, skip);
        }
        return dp[nums.length];
    }
}
```
