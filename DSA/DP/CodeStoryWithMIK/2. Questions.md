
### 1. Fibonacci Series

###### Recursive
```java
f(int n)
{
	// base case
	if (n == 0 or n == 1)
	{
		return n;
	}	
	return f(n-1) + f(n-2);
}
```

> Dimensions of DP array = no. of changing variables


###### Memoization
```java
Integer[] dp;
f(int n)
{
	// base case
	if (n == 0 or n == 1)
	{
		return n;
	}	
	if (dp[n] != null)
	{
		return dp[n];
	}
	return dp[n] = f(n-1) + f(n-2);
}
```


###### Bottom Up
```java
class Solution {
    Integer[] dp;
    public int fib(int n) {
        dp = new Integer[n+1];
        return f(n);
    }

    private int f(int n)
    {
        // base case
        if (n <= 1)
        {
            return n;
        }
        dp[0] = 0;
        dp[1] = 1;
        
        for (int i = 2; i <= n; i++)
        {
            dp[i] = dp[i-1] + dp[i-2];
        }

        return dp[n];
    }
}
```

###### Constant Space
```java
class Solution {
    public int fib(int n) {
        return f(n);
    }

    private int f(int n)
    {
        // base case
        if (n <= 1)
        {
            return n;
        }
        int prev1 = 0;
        int prev2 = 1;
        
        for (int i = 2; i <= n; i++)
        {
            int curr = prev1 + prev2;
            prev1 = prev2;
            prev2 = curr;
        }

        return prev2;
    }
}
```


### 2. [Climbing Stairs](https://leetcode.com/problems/climbing-stairs/)

###### Memoization - Top Down

```java
class Solution {
    Integer[] dp;
    public int climbStairs(int n) {
        dp = new Integer[n+1];
        return solve(n);
    }

    private int solve(int n)
    {
        // base case
        if (n < 0)  return 0;
        if (n == 0) return 1;
        if (dp[n] != null)
        {
            return dp[n];
        }

        int s1 = solve(n-1);
        int s2 = solve(n-2);

        return dp[n] = s1 + s2;
    }
}
```

###### Bottom Up

```java
class Solution {
    Integer[] dp;
    public int climbStairs(int n) {
        dp = new Integer[n+1];
        return solve(n);
    }

    private int solve(int n)
    {
        // base case
        if (n == 0 || n == 1 || n == 2)
            return n;

        dp[0] = 0;
        dp[1] = 1;
        dp[2] = 2;

        for (int i = 3; i <= n; i++)
        {
            dp[i] = dp[i-1] + dp[i-2];
        }
        return dp[n];
    }
}
```

## 3. [House Robber](https://leetcode.com/problems/house-robber/)
###### Recursive
```java
class Solution {
    public int rob(int[] nums) {
        return solve(0, nums);
    }
    private int solve(int i, int[] nums)
    {
        // base case
        if (i >= nums.length)
        {
            return 0;
        }
        int steal = nums[i] + solve(i+2, nums);
        int skip = solve(i+1, nums);

        return Math.max(steal, skip);
    }
}
```

```java
class Solution {
```

```java
        dp = new Integer[nums.length+1];
        return solve(nums);
    }
    private int solve(int[] nums)
    {
        // base case
        if (nums.length == 1)
        {
            return nums[0];
        }

        dp[0] = 0;
        dp[1] = nums[0];

        for (int i = 2; i<= nums.length; i++)
        {
            int steal = nums[i-1] + dp[i-2];
            int skip = dp[i-1];
            dp[i] = Math.max(steal, skip);
        }
        return dp[nums.length];
    }
}
```

## 4. [House Robber II](https://leetcode.com/problems/house-robber-ii/)
###### Recursive and Memoization
```java
class Solution {
    Integer[] dp1, dp2;

    public int rob(int[] nums) {
        int n = nums.length;
        
        // Edge cases
        if (n == 0) return 0; // No house to rob
        if (n == 1) return nums[0]; // Only one house to rob

        // Initialize two dp arrays for the two cases
        dp1 = new Integer[n];
        dp2 = new Integer[n];
        
        // Case 1: Exclude the last house
        int c1 = solve(0, nums, n - 2, dp1);
        // Case 2: Exclude the first house
        int c2 = solve(1, nums, n - 1, dp2);
        
        // Return the maximum of the two cases
        return Math.max(c1, c2);
    }

    private int solve(int ind, int[] nums, int n, Integer[] dp) {
        // Base case
        if (ind > n) {
            return 0;
        }
        if (dp[ind] != null) {
            return dp[ind];
        }

        // Calculate the maximum amount of money that can be robbed
        int steal = nums[ind] + solve(ind + 2, nums, n, dp);
        int skip = solve(ind + 1, nums, n, dp);

        // Store the result in dp array
        return dp[ind] = Math.max(steal, skip);
    }
}
```
###### Bottom Up
```java
class Solution {
    Integer[] dp1, dp2;

    public int rob(int[] nums) {
        int n = nums.length;
        if (n == 0) return 0; // No house to rob
        if (n == 1) return nums[0]; // Only one house to rob

        // Initialize two dp arrays for the two cases
        dp1 = new Integer[n];
        dp2 = new Integer[n];

        // Initialize for case 1: Rob the first house, don't rob the last house
        dp1[0] = nums[0];
        dp1[1] = Math.max(nums[0], nums[1]);

        // Initialize for case 2: Don't rob the first house, consider the last house
        dp2[0] = 0;
        dp2[1] = nums[1];

        // Fill dp arrays
        for (int i = 2; i < n; i++) {
            dp1[i] = Math.max(dp1[i - 1], dp1[i - 2] + nums[i]);
        }

        for (int i = 2; i < n; i++) {
            dp2[i] = Math.max(dp2[i - 1], dp2[i - 2] + nums[i]);
        }

        // Since we are considering a circular array, for dp1, we don't consider the last element
        // For dp2, we don't consider the first element
        return Math.max(dp1[n - 2], dp2[n - 1]);
    }
}
```
